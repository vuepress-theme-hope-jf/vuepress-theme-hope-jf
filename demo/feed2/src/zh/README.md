## 使用插件

### 安装

```bash
pnpm add -D vuepress-plugin-feed2
```

### 使用

```ts
// .vuepress/config.ts
import { feedPlugin } from "vuepress-plugin-feed2";

export default {
  plugins: [
    feedPlugin({
      // 插件选项
    }),
  ],
};
```
