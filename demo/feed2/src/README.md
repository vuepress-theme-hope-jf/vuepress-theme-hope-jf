## How to use

### Install

```bash
pnpm add -D vuepress-plugin-feed2
```

### Usage

```ts
// .vuepress/config.ts
import { feedPlugin } from "vuepress-plugin-feed2";

export default {
  plugins: [
    feedPlugin({
      // your options
    }),
  ],
};
```
